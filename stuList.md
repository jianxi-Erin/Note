'''c
#define MAX_SIZE 100  // 最大学生数量

// 学生结构体
typedef struct {
    int id;          // 学号
    char name[50];   // 姓名
    float score;     // 分数
} Student;

// 顺序表
typedef struct {
    Student data[MAX_SIZE];  // 存储学生的数组
    int size;                // 当前学生数量
} StudentList;

// 初始化学生顺序表
void initList(StudentList *list) {
    list->size = 0;
}
